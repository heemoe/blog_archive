用正则表达式判断字符串是否符合条件

ZSMS_HEE
2015-12-30
Objective C, iOS

@regex 正则表达
@string 字符串

- (BOOL)boolWithRegex:(NSString *)regex string:(NSString *)str{
    NSPredicate *pred = [NSPredicate predicateWithFormat:@"SELF MATCHES %@",regex];
    BOOL isMatch = [pred evaluateWithObject:str];
    return isMatch;
}
