记录一个cell 侧滑的pan手势监听代码

ZSMS_HEE
2015-12-18 
- (void)panActon:(UIPanGestureRecognizer *)pan{

    if (pan.state == UIGestureRecognizerStateBegan) {
        // 开始
        _originX = self.x;
    }
    if (pan.state == UIGestureRecognizerStateChanged) {
       CGPoint cur = [pan translationInView:self];
        if (_originX + cur.x + MaxOffset &lt; 0 || cur.x + _originX &gt; 0) {
            return;
        }
        self.x = _originX + cur.x;
    }
    if (pan.state == UIGestureRecognizerStateEnded){
        CGPoint point = [pan translationInView:self];
        if(_originX + MaxOffset + point.x &gt; MaxOffset / 2)
        {
            [UIView animateWithDuration:animDur animations:^{
                self.x = 0;
            }];
        }
        else
        {
            [UIView animateWithDuration:animDur animations:^{
                self.x = -MaxOffset;
            }];
        }
    }
}
