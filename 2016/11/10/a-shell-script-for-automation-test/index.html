#!/bin/sh
#@author hee
LogParse(){
  echo 'Log Parse Begin'
  now=$(date +%Y-%m-%d-%T)
  path=TestLogs
  fullLogPath=$path/$now.log
  errorLogPath=$path/error_$now.log
  echo $(pwd)/$fullLogPath
  echo $(pwd)/$errorLogPath
  if [ ! -d $path ]; then
  mkdir $path
  fi
  xcodebuild test -project mobile.xcodeproj -scheme mobile -destination "$1" &gt; $fullLogPath
  # headLineNum=$(awk '/========== Detailed Exception ==========/{ print NR;}' $fullLogPath) #find a head
  # endLineNum=$(awk '/ failed \(/{ print NR;}' $fullLogPath) #find end line
  headLineNum=$(awk '/Exception: AssertionFailedException/{ print NR;}' $fullLogPath) #find a head
  endLineNum=$(awk '/Application window hierarchy \(/{ print NR;}' $fullLogPath) #find end line
  hNums=(${headLineNum//\ / })
  eNums=(${endLineNum//\ / })
  echo $hNums
  if [ ${#hNums} = ${#eNums} ]; then #heads equal ends
    echo ${#hNums}
    count=${#eNums[@]}
    for ((i=0;i&lt;count;i++))
      do
        for ((n=${hNums[$i]};n&lt;=${eNums[$i]};n++))
          do
            sed "${n}q;d" $fullLogPath &gt;&gt; $errorLogPath # write error content
          done
      done
  elif [ ${#hNums}=0 -a ${#eNums}=0 ] ;then #heads and ends are equal 0
      echo 'error'
  else 
    echo 'read line numbers error'
  fi
  tail -n 2 $fullLogPath &gt;&gt; $errorLogPath  # write 2 line of end error
}

if [ "$1" = "" ]; then
devices=("platform=iOS Simulator,OS=9.3,name=iPhone 5"
"platform=iOS Simulator,OS=9.3,name=iPhone 5s"
"platform=iOS Simulator,OS=9.3,name=iPhone 6s"
"platform=iOS Simulator,OS=9.3,name=iPhone 6s Plus")
for i in "${devices[@]}"
do
  echo $i
  LogParse "$i"
done 
exit
fi

if [ ! -e $1 ]; then
echo 'run test with xlsx'
echo '  earlgreyTool.sh [XLSXFILE]    eg. earlgreyTool.sh ~/example.xlsx ' 
echo 'run test'
echo '  earlgreyTool.sh run '
exit;
fi

# process sheet 1
echo ##############process sheet 1################
xlsx2csv $1 data.csv -s 1
if [ ! $? -eq 0 ]; then
    echo 'You have not install xlsx2csv !'
    echo 'Please execute sudo easy_install xlsx2csv or pip install xlsx2csv.'
    exit
fi
sql='csv2sql.sql'
if [ -e $sql ]; then
rm -rf $sql
echo 'removed exists .sql file'
fi 
touch $sql
echo 'DROP TABLE DATA;' &gt;&gt; $sql
echo 'CREATE TABLE  data(caseId,dataId,inputData,expectation,mark);' &gt;&gt; $sql
echo '.mode csv' &gt;&gt; $sql
echo '.import data.csv data' &gt;&gt; $sql
echo "DELETE FROM data WHERE caseId='caseId';" &gt;&gt; $sql
echo '.header on' &gt;&gt; $sql
echo '.mode column' &gt;&gt; $sql
echo 'SELECT * FROM data;' &gt;&gt; $sql
sqlite3 data.sqlite &lt; $sql
rm $sql
rm data.csv

echo ##############process sheet 2################
xlsx2csv $1 data.csv -s 2
sql='csv2sql.sql'
touch $sql
echo 'DROP TABLE COM;' &gt;&gt; $sql
echo 'CREATE TABLE com(key,value);' &gt;&gt; $sql
echo '.mode csv' &gt;&gt; $sql
echo '.import data.csv com' &gt;&gt; $sql
echo "DELETE FROM com WHERE key='key';" &gt;&gt; $sql
echo '.header on' &gt;&gt; $sql
echo '.mode column' &gt;&gt; $sql
echo 'SELECT * FROM com;' &gt;&gt; $sql
sqlite3 data.sqlite &lt; $sql
rm $sql
rm data.csv
